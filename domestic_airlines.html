df2 <- df %>% filter((Airline == 'Jetstar' | Airline == 'Qantas'),
                     Route != 'All Ports-All Ports')

df2 %>%
  group_by(Airline, Year, Departing_Port) %>%
  summarise(n = n()) %>%
  ungroup %>%
    ggplot(aes(Departing_Port, n, color = Airline)) +
    geom_boxplot() +
    theme(axis.text.x=element_text(angle=45, hjust=1))
